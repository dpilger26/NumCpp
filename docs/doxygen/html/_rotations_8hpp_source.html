155  static NdArray<double> angularVelocity(const Quaternion& inQuat1, const Quaternion& inQuat2, double inTime)
156  {
157  NdArray<double> q0 = inQuat1.toNdArray();
158  NdArray<double> q1 = inQuat2.toNdArray();
159 
160  NdArray<double> qDot = q1 - q0;
161  qDot /= inTime;
162 
163  NdArray<double> eyeTimesScalar(3);
164  eyeTimesScalar.zeros();
165  eyeTimesScalar(0, 0) = inQuat2.s();
166  eyeTimesScalar(1, 1) = inQuat2.s();
167  eyeTimesScalar(2, 2) = inQuat2.s();
168 
169  NdArray<double> epsilonHat = Linalg::hat<double>(inQuat2.i(), inQuat2.j(), inQuat2.k());
170  NdArray<double> q(4, 3);
171  q.put(Slice(0, 3), Slice(0, 3), eyeTimesScalar + epsilonHat);
172  q(3, 0) = -inQuat2.i();
173  q(3, 1) = -inQuat2.j();
174  q(3, 2) = -inQuat2.k();
175 
176  NdArray<double> omega = q.transpose().dot<double>(qDot.transpose());
177  return std::move(omega *= 2);
178  }

278  static Quaternion fromDCM(const NdArray<dtype>& inDcm)
279  {
280  Shape inShape = inDcm.shape();
281  if (!(inShape.rows == 3 && inShape.cols == 3))
282  {
283  throw std::invalid_argument("ERROR: Rotations::Quaternion::fromDcm: input direction cosine matrix must have shape = (3,3).");
284  }
285 
286  NdArray<double> dcm = inDcm.astype<double>();
287 
288  NdArray<double> checks(1, 4);
289  checks[0] = dcm(0, 0) + dcm(1, 1) + dcm(2, 2);
290  checks[1] = dcm(0, 0) - dcm(1, 1) + dcm(2, 2);
291  checks[2] = dcm(0, 0) - dcm(1, 1) - dcm(2, 2);
292  checks[3] = dcm(0, 0) + dcm(1, 1) - dcm(2, 2);
293 
294  uint32 maxIdx = argmax(checks).item();
295 
296  double q0 = 0;
297  double q1 = 0;
298  double q2 = 0;
299  double q3 = 0;
300 
301  switch (maxIdx)
302  {
303  case 0:
304  {
305  q3 = 0.5 * std::sqrt(1 + dcm(0, 0) + dcm(1, 1) + dcm(2, 2));
306  q0 = (dcm(1, 2) - dcm(2, 1)) / (4 * q3);
307  q1 = (dcm(2, 0) - dcm(0, 2)) / (4 * q3);
308  q2 = (dcm(0, 1) - dcm(1, 0)) / (4 * q3);
309 
310  break;
311  }
312  case 1:
313  {
314  q2 = 0.5 * std::sqrt(1 - dcm(0, 0) - dcm(1, 1) + dcm(2, 2));
315  q0 = (dcm(0, 2) + dcm(2, 0)) / (4 * q2);
316  q1 = (dcm(1, 2) + dcm(2, 1)) / (4 * q2);
317  q3 = (dcm(0, 1) - dcm(1, 0)) / (4 * q2);
318 
319  break;
320  }
321  case 2:
322  {
323  q0 = 0.5 * std::sqrt(1 + dcm(0, 0) - dcm(1, 1) - dcm(2, 2));
324  q1 = (dcm(0, 1) + dcm(1, 0)) / (4 * q0);
325  q2 = (dcm(0, 2) + dcm(2, 0)) / (4 * q0);
326  q3 = (dcm(1, 2) - dcm(2, 1)) / (4 * q0);
327 
328  break;
329  }
330  case 3:
331  {
332  q1 = 0.5 * std::sqrt(1 - dcm(0, 0) + dcm(1, 1) - dcm(2, 2));
333  q0 = (dcm(0, 1) + dcm(1, 0)) / (4 * q1);
334  q2 = (dcm(1, 2) + dcm(2, 1)) / (4 * q1);
335  q3 = (dcm(2, 0) - dcm(0, 2)) / (4 * q1);
336 
337  break;
338  }
339  }
340 
341  return Quaternion(q0, q1, q2, q3);
342  }

444  static Quaternion slerp(const Quaternion& inQuat1, const Quaternion& inQuat2, double inPercent)
445  {
446  if (inPercent < 0 || inPercent > 1)
447  {
448  throw std::invalid_argument("ERROR: Rotations::Quaternion::slerp: input percent must be of the range [0, 1]");
449  }
450 
451  if (inPercent == 0)
452  {
453  return inQuat1;
454  }
455  else if (inPercent == 1)
456  {
457  return inQuat2;
458  }
459 
460  double dotProduct = dot<double, double>(inQuat1.toNdArray(), inQuat2.toNdArray()).item();
461 
462  // If the dot product is negative, the quaternions
463  // have opposite handed-ness and slerp won't take
464  // the shorter path. Fix by reversing one quaternion.
465  Quaternion quat1Copy(inQuat1);
466  if (dotProduct < 0.0)
467  {
468  quat1Copy *= -1;
469  dotProduct *= -1;
470  }
471 
472  const double DOT_THRESHOLD = 0.9995;
473  if (dotProduct > DOT_THRESHOLD) {
474  // If the inputs are too close for comfort, linearly interpolate
475  // and normalize the result.
476  return nlerp(inQuat1, inQuat2, inPercent);
477  }
478 
479  dotProduct = clip(dotProduct, -1.0, 1.0); // Robustness: Stay within domain of acos()
480  double theta0 = std::acos(dotProduct); // angle between input vectors
481  double theta = theta0 * inPercent; // angle between v0 and result
482 
483  double s0 = std::cos(theta) - dotProduct * std::sin(theta) / std::sin(theta0); // == sin(theta_0 - theta) / sin(theta_0)
484  double s1 = std::sin(theta) / std::sin(theta0);
485 
486  NdArray<double> interpQuat = (inQuat1.toNdArray() * s0) + (inQuat2.toNdArray() * s1);
487  return Quaternion(interpQuat);
488  }

526  NdArray<double> toDCM() const
527  {
528  NdArray<double> dcm(3);
529 
530  double q0 = i();
531  double q1 = j();
532  double q2 = k();
533  double q3 = s();
534 
535  dcm(0, 0) = Utils::sqr(q3) + Utils::sqr(q0) - Utils::sqr(q1) - Utils::sqr(q2);
536  dcm(0, 1) = 2 * (q0 * q1 + q3 * q2);
537  dcm(0, 2) = 2 * (q0 * q2 - q3 * q1);
538  dcm(1, 0) = 2 * (q0 * q1 - q3 * q2);
539  dcm(1, 1) = Utils::sqr(q3) - Utils::sqr(q0) + Utils::sqr(q1) - Utils::sqr(q2);;
540  dcm(1, 2) = 2 * (q1 * q2 + q3 * q0);
541  dcm(2, 0) = 2 * (q0 * q2 + q3 * q1);
542  dcm(2, 1) = 2 * (q1 * q2 - q3 * q0);
543  dcm(2, 2) = Utils::sqr(q3) - Utils::sqr(q0) - Utils::sqr(q1) + Utils::sqr(q2);;
544 
545  return std::move(dcm);
546  }
