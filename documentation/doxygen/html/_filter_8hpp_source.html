701  static NdArray<dtype> convolve(const NdArray<dtype>& inImageArray, uint32 inSize,
702  const NdArray<dtype>& inWeights, Filter::Boundary::Mode inMode = Filter::Boundary::REFLECT, dtype inConstantValue = 0)
703  {
704  if (inWeights.size() != Utils<uint32>::sqr(inSize))
705  {
706  throw std::invalid_argument("ERROR: NumC::Filters::convolve: input weights do no match input kernal size.");
707  }
708 
709  NdArray<dtype> arrayWithBoundary = addBoundary(inImageArray, inMode, inSize, inConstantValue);
710  NdArray<dtype> output(inImageArray.shape());
711 
712  NdArray<dtype> weightsFlat = Methods<dtype>::rot90(inWeights, 2).flatten();
713  Shape inShape = inImageArray.shape();
714  uint32 boundarySize = inSize / 2; // integer division
715  uint32 endPointRow = boundarySize + inShape.rows;
716  uint32 endPointCol = boundarySize + inShape.cols;
717 
718  for (uint32 row = boundarySize; row < endPointRow; ++row)
719  {
720  for (uint32 col = boundarySize; col < endPointCol; ++col)
721  {
722  NdArray<dtype> window = arrayWithBoundary(Slice(row - boundarySize, row + boundarySize + 1),
723  Slice(col - boundarySize, col + boundarySize + 1)).flatten();
724 
725  output(row - boundarySize, col - boundarySize) = Methods<dtype>::dot(window, weightsFlat).item();
726  }
727  }
728 
729  return std::move(output);
730  }
