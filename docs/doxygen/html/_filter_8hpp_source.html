959  template<typename dtype>
960  NdArray<dtype> gaussianFilter(const NdArray<dtype>& inImageArray, double inSigma,
961  Boundary inBoundaryType, dtype inConstantValue)
962  {
963  if (inSigma <= 0)
964  {
965  std::string errStr = "ERROR: NC::Filters::gaussianFilter: input sigma value must be greater than zero.";
966  std::cerr << errStr << std::endl;
967  throw std::invalid_argument(errStr);
968  }
969 
970  // calculate the kernel size based off of the input sigma value
971  const uint32 MIN_KERNEL_SIZE = 5;
972  uint32 kernelSize = std::max(static_cast<uint32>(std::ceil(inSigma * 2.0 * 4.0)), MIN_KERNEL_SIZE); // 4 standard deviations
973  if (kernelSize % 2 == 0)
974  {
975  ++kernelSize; // make sure the kernel is an odd size
976  }
977 
978  const double kernalHalfSize = static_cast<double>(kernelSize / 2); // integer division
979 
980  // calculate the gaussian kernel
981  NdArray<double> kernel(kernelSize);
982  for (double row = 0; row < kernelSize; ++row)
983  {
984  for (double col = 0; col < kernelSize; ++col)
985  {
986  kernel(static_cast<uint32>(row), static_cast<uint32>(col)) =
987  gaussian(row - kernalHalfSize, col - kernalHalfSize, inSigma);
988  }
989  }
990 
991  // normalize the kernel
992  kernel /= kernel.template sum<double>().item();
993 
994  // perform the convolution
995  NdArray<dtype> output = convolve(inImageArray.template astype<double>(),
996  kernelSize,
997  kernel,
998  inBoundaryType,
999  inConstantValue).template astype<dtype>();
1000 
1001  return output;
1002  }
