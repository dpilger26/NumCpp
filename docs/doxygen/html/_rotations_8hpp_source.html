160  static NdArray<double> angularVelocity(const Quaternion& inQuat1, const Quaternion& inQuat2, double inTime)
161  {
162  NdArray<double> q0 = inQuat1.toNdArray();
163  NdArray<double> q1 = inQuat2.toNdArray();
164 
165  NdArray<double> qDot = q1 - q0;
166  qDot /= inTime;
167 
168  NdArray<double> eyeTimesScalar(3);
169  eyeTimesScalar.zeros();
170  eyeTimesScalar(0, 0) = inQuat2.s();
171  eyeTimesScalar(1, 1) = inQuat2.s();
172  eyeTimesScalar(2, 2) = inQuat2.s();
173 
174  NdArray<double> epsilonHat = linalg::hat<double>(inQuat2.i(), inQuat2.j(), inQuat2.k());
175  NdArray<double> q(4, 3);
176  q.put(Slice(0, 3), Slice(0, 3), eyeTimesScalar + epsilonHat);
177  q(3, 0) = -inQuat2.i();
178  q(3, 1) = -inQuat2.j();
179  q(3, 2) = -inQuat2.k();
180 
181  NdArray<double> omega = q.transpose().template dot<double>(qDot.transpose());

283  static Quaternion fromDCM(const NdArray<dtype>& inDcm)
284  {
285  const Shape inShape = inDcm.shape();
286  if (!(inShape.rows == 3 && inShape.cols == 3))
287  {
288  std::string errStr = "ERROR: rotations::Quaternion::fromDcm: input direction cosine matrix must have shape = (3,3).";
289  std::cerr << errStr << std::endl;
290  throw std::invalid_argument(errStr);
291  }
292 
293  NdArray<double> dcm = inDcm.template astype<double>();
294 
295  NdArray<double> checks(1, 4);
296  checks[0] = dcm(0, 0) + dcm(1, 1) + dcm(2, 2);
297  checks[1] = dcm(0, 0) - dcm(1, 1) + dcm(2, 2);
298  checks[2] = dcm(0, 0) - dcm(1, 1) - dcm(2, 2);
299  checks[3] = dcm(0, 0) + dcm(1, 1) - dcm(2, 2);
300 
301  uint32 maxIdx = argmax(checks).item();
302 
303  double q0 = 0;
304  double q1 = 0;
305  double q2 = 0;
306  double q3 = 0;
307 
308  switch (maxIdx)
309  {
310  case 0:
311  {
312  q3 = 0.5 * std::sqrt(1 + dcm(0, 0) + dcm(1, 1) + dcm(2, 2));
313  q0 = (dcm(1, 2) - dcm(2, 1)) / (4 * q3);
314  q1 = (dcm(2, 0) - dcm(0, 2)) / (4 * q3);
315  q2 = (dcm(0, 1) - dcm(1, 0)) / (4 * q3);
316 
317  break;
318  }
319  case 1:
320  {
321  q2 = 0.5 * std::sqrt(1 - dcm(0, 0) - dcm(1, 1) + dcm(2, 2));
322  q0 = (dcm(0, 2) + dcm(2, 0)) / (4 * q2);
323  q1 = (dcm(1, 2) + dcm(2, 1)) / (4 * q2);
324  q3 = (dcm(0, 1) - dcm(1, 0)) / (4 * q2);
325 
326  break;
327  }
328  case 2:
329  {
330  q0 = 0.5 * std::sqrt(1 + dcm(0, 0) - dcm(1, 1) - dcm(2, 2));
331  q1 = (dcm(0, 1) + dcm(1, 0)) / (4 * q0);
332  q2 = (dcm(0, 2) + dcm(2, 0)) / (4 * q0);
333  q3 = (dcm(1, 2) - dcm(2, 1)) / (4 * q0);
334 
335  break;
336  }
337  case 3:
338  {
339  q1 = 0.5 * std::sqrt(1 - dcm(0, 0) + dcm(1, 1) - dcm(2, 2));
340  q0 = (dcm(0, 1) + dcm(1, 0)) / (4 * q1);
341  q2 = (dcm(1, 2) + dcm(2, 1)) / (4 * q1);
342  q3 = (dcm(2, 0) - dcm(0, 2)) / (4 * q1);
343 
344  break;
345  }
346  }
347 
348  return Quaternion(q0, q1, q2, q3);
349  }

455  static Quaternion slerp(const Quaternion& inQuat1, const Quaternion& inQuat2, double inPercent)
456  {
457  if (inPercent < 0 || inPercent > 1)
458  {
459  std::string errStr = "ERROR: rotations::Quaternion::slerp: input percent must be of the range [0, 1]";
460  std::cerr << errStr << std::endl;
461  throw std::invalid_argument(errStr);
462  }
463 
464  if (inPercent == 0)
465  {
466  return inQuat1;
467  }
468  else if (inPercent == 1)
469  {
470  return inQuat2;
471  }
472 
473  double dotProduct = dot<double>(inQuat1.toNdArray(), inQuat2.toNdArray()).item();
474 
475  // If the dot product is negative, the quaternions
476  // have opposite handed-ness and slerp won't take
477  // the shorter path. Fix by reversing one quaternion.
478  Quaternion quat1Copy(inQuat1);
479  if (dotProduct < 0.0)
480  {
481  quat1Copy *= -1;
482  dotProduct *= -1;
483  }
484 
485  const double DOT_THRESHOLD = 0.9995;
486  if (dotProduct > DOT_THRESHOLD) {
487  // If the inputs are too close for comfort, linearly interpolate
488  // and normalize the result.
489  return nlerp(inQuat1, inQuat2, inPercent);
490  }
491 
492  dotProduct = clip(dotProduct, -1.0, 1.0); // Robustness: Stay within domain of acos()
493  const double theta0 = std::acos(dotProduct); // angle between input vectors
494  const double theta = theta0 * inPercent; // angle between v0 and result
495 
496  const double s0 = std::cos(theta) - dotProduct * std::sin(theta) / std::sin(theta0); // == sin(theta_0 - theta) / sin(theta_0)
497  const double s1 = std::sin(theta) / std::sin(theta0);
498 
499  NdArray<double> interpQuat = (inQuat1.toNdArray() * s0) + (inQuat2.toNdArray() * s1);
500  return Quaternion(interpQuat);
501  }

539  NdArray<double> toDCM() const
540  {
541  NdArray<double> dcm(3);
542 
543  double q0 = i();
544  double q1 = j();
545  double q2 = k();
546  double q3 = s();
547 
548  dcm(0, 0) = utils::sqr(q3) + utils::sqr(q0) - utils::sqr(q1) - utils::sqr(q2);
549  dcm(0, 1) = 2 * (q0 * q1 + q3 * q2);
550  dcm(0, 2) = 2 * (q0 * q2 - q3 * q1);
551  dcm(1, 0) = 2 * (q0 * q1 - q3 * q2);
552  dcm(1, 1) = utils::sqr(q3) - utils::sqr(q0) + utils::sqr(q1) - utils::sqr(q2);;
553  dcm(1, 2) = 2 * (q1 * q2 + q3 * q0);
554  dcm(2, 0) = 2 * (q0 * q2 + q3 * q1);
555  dcm(2, 1) = 2 * (q1 * q2 - q3 * q0);
556  dcm(2, 2) = utils::sqr(q3) - utils::sqr(q0) - utils::sqr(q1) + utils::sqr(q2);;
557 
558  return dcm;
559  }
