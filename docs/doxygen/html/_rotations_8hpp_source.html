175  static NdArray<double> angularVelocity(const Quaternion& inQuat1, const Quaternion& inQuat2, double inTime)
176  {
177  NdArray<double> q0 = inQuat1.toNdArray();
178  NdArray<double> q1 = inQuat2.toNdArray();
179 
180  NdArray<double> qDot = q1 - q0;
181  qDot /= inTime;
182 
183  NdArray<double> eyeTimesScalar(3);
184  eyeTimesScalar.zeros();
185  eyeTimesScalar(0, 0) = inQuat2.s();
186  eyeTimesScalar(1, 1) = inQuat2.s();
187  eyeTimesScalar(2, 2) = inQuat2.s();
188 
189  NdArray<double> epsilonHat = Linalg<double>::hat(inQuat2.i(), inQuat2.j(), inQuat2.k());
190  NdArray<double> q(4, 3);
191  q.put(Slice(0, 3), Slice(0, 3), eyeTimesScalar + epsilonHat);
192  q(3, 0) = -inQuat2.i();
193  q(3, 1) = -inQuat2.j();
194  q(3, 2) = -inQuat2.k();
195 
196  NdArray<double> omega = q.transpose().dot<double>(qDot.transpose());
197  return std::move(omega *= 2);
198  }

310  static Quaternion fromDCM(const NdArray<dtype>& inDcm)
311  {
312  Shape inShape = inDcm.shape();
313  if (!(inShape.rows == 3 && inShape.cols == 3))
314  {
315  throw std::invalid_argument("ERROR: Rotations::Quaternion::fromDcm: input direction cosine matrix must have shape = (3,3).");
316  }
317 
318  NdArray<double> dcm = inDcm.astype();
319 
320  NdArray<double> checks(1, 4);
321  checks[0] = dcm(0, 0) + dcm(1, 1) + dcm(2, 2);
322  checks[1] = dcm(0, 0) - dcm(1, 1) + dcm(2, 2);
323  checks[2] = dcm(0, 0) - dcm(1, 1) - dcm(2, 2);
324  checks[3] = dcm(0, 0) + dcm(1, 1) - dcm(2, 2);
325 
326  uint32 maxIdx = Methods<double>::argmax(checks).item();
327 
328  double q0 = 0;
329  double q1 = 0;
330  double q2 = 0;
331  double q3 = 0;
332 
333  switch (maxIdx)
334  {
335  case 0:
336  {
337  q3 = 0.5 * std::sqrt(1 + dcm(0, 0) + dcm(1, 1) + dcm(2, 2));
338  q0 = (dcm(1, 2) - dcm(2, 1)) / (4 * q3);
339  q1 = (dcm(2, 0) - dcm(0, 2)) / (4 * q3);
340  q2 = (dcm(0, 1) - dcm(1, 0)) / (4 * q3);
341 
342  break;
343  }
344  case 1:
345  {
346  q2 = 0.5 * std::sqrt(1 - dcm(0, 0) - dcm(1, 1) + dcm(2, 2));
347  q0 = (dcm(0, 2) + dcm(2, 0)) / (4 * q2);
348  q1 = (dcm(1, 2) + dcm(2, 1)) / (4 * q2);
349  q3 = (dcm(0, 1) - dcm(1, 0)) / (4 * q2);
350 
351  break;
352  }
353  case 2:
354  {
355  q0 = 0.5 * std::sqrt(1 + dcm(0, 0) - dcm(1, 1) - dcm(2, 2));
356  q1 = (dcm(0, 1) + dcm(1, 0)) / (4 * q0);
357  q2 = (dcm(0, 2) + dcm(2, 0)) / (4 * q0);
358  q3 = (dcm(1, 2) - dcm(2, 1)) / (4 * q0);
359 
360  break;
361  }
362  case 3:
363  {
364  q1 = 0.5 * std::sqrt(1 - dcm(0, 0) + dcm(1, 1) - dcm(2, 2));
365  q0 = (dcm(0, 1) + dcm(1, 0)) / (4 * q1);
366  q2 = (dcm(1, 2) + dcm(2, 1)) / (4 * q1);
367  q3 = (dcm(2, 0) - dcm(0, 2)) / (4 * q1);
368 
369  break;
370  }
371  }
372 
373  return Quaternion(q0, q1, q2, q3);
374  }

483  static Quaternion slerp(const Quaternion& inQuat1, const Quaternion& inQuat2, double inPercent)
484  {
485  if (inPercent < 0 || inPercent > 1)
486  {
487  throw std::invalid_argument("ERROR: Rotations::Quaternion::slerp: input percent must be of the range [0, 1]");
488  }
489 
490  if (inPercent == 0)
491  {
492  return inQuat1;
493  }
494  else if (inPercent == 1)
495  {
496  return inQuat2;
497  }
498 
499  double dotProduct = Methods<double>::dot<double>(inQuat1.toNdArray(), inQuat2.toNdArray()).item();
500 
501  // If the dot product is negative, the quaternions
502  // have opposite handed-ness and slerp won't take
503  // the shorter path. Fix by reversing one quaternion.
504  Quaternion quat1Copy(inQuat1);
505  if (dotProduct < 0.0)
506  {
507  quat1Copy *= -1;
508  dotProduct *= -1;
509  }
510 
511  const double DOT_THRESHOLD = 0.9995;
512  if (dotProduct > DOT_THRESHOLD) {
513  // If the inputs are too close for comfort, linearly interpolate
514  // and normalize the result.
515  return nlerp(inQuat1, inQuat2, inPercent);
516  }
517 
518  dotProduct = Methods<double>::clip(dotProduct, -1.0, 1.0); // Robustness: Stay within domain of acos()
519  double theta0 = std::acos(dotProduct); // angle between input vectors
520  double theta = theta0 * inPercent; // angle between v0 and result
521 
522  double s0 = std::cos(theta) - dotProduct * std::sin(theta) / std::sin(theta0); // == sin(theta_0 - theta) / sin(theta_0)
523  double s1 = std::sin(theta) / std::sin(theta0);
524 
525  NdArray<double> interpQuat = (inQuat1.toNdArray() * s0) + (inQuat2.toNdArray() * s1);
526  return Quaternion(interpQuat);
527  }

569  NdArray<double> toDCM() const
570  {
571  NdArray<double> dcm(3);
572 
573  double q0 = i();
574  double q1 = j();
575  double q2 = k();
576  double q3 = s();
577 
578  dcm(0, 0) = Utils<double>::sqr(q3) + Utils<double>::sqr(q0) - Utils<double>::sqr(q1) - Utils<double>::sqr(q2);
579  dcm(0, 1) = 2 * (q0 * q1 + q3 * q2);
580  dcm(0, 2) = 2 * (q0 * q2 - q3 * q1);
581  dcm(1, 0) = 2 * (q0 * q1 - q3 * q2);
582  dcm(1, 1) = Utils<double>::sqr(q3) - Utils<double>::sqr(q0) + Utils<double>::sqr(q1) - Utils<double>::sqr(q2);;
583  dcm(1, 2) = 2 * (q1 * q2 + q3 * q0);
584  dcm(2, 0) = 2 * (q0 * q2 + q3 * q1);
585  dcm(2, 1) = 2 * (q1 * q2 - q3 * q0);
586  dcm(2, 2) = Utils<double>::sqr(q3) - Utils<double>::sqr(q0) - Utils<double>::sqr(q1) + Utils<double>::sqr(q2);;
587 
588  return std::move(dcm);
589  }
