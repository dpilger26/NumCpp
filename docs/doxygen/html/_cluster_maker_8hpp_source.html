291  void runClusterMaker()
292  {
293  uint32 clusterId = 0;
294 
295  for (auto& currentPixel : xcdsVec_)
296  {
297  // not already visited
298  if (currentPixel.clusterId == -1)
299  {
300  Cluster<dtype> newCluster(clusterId); // a new cluster
301  currentPixel.clusterId = clusterId;
302  newCluster.addPixel(currentPixel); // assign pixel to cluster
303 
304  // get the neighbors
305  std::vector<uint32> neighborIds;
306  findNeighborXcds(currentPixel, neighborIds);
307  if (neighborIds.empty())
308  {
309  clusters_.push_back(newCluster);
310  ++clusterId;
311  continue;
312  }
313 
314  // loop through the neighbors
315  for (uint32 neighborsIdx = 0; neighborsIdx < neighborIds.size(); ++neighborsIdx)
316  {
317  Pixel<dtype>& currentNeighborPixel = xcdsVec_[neighborIds[neighborsIdx]];
318 
319  // go to neighbors
320  std::vector<uint32> newNeighborIds;
321  findNeighborXcds(currentNeighborPixel, newNeighborIds);
322 
323  // loop through the new neighbors and add them to neighbors
324  for (auto newNeighborId : newNeighborIds)
325  {
326  // not already in neighbors
327  if (std::find(neighborIds.begin(), neighborIds.end(), newNeighborId) == neighborIds.end())
328  {
329  neighborIds.push_back(newNeighborId);
330  }
331  }
332 
333  // not already assigned to a cluster
334  if (currentNeighborPixel.clusterId == -1)
335  {
336  currentNeighborPixel.clusterId = clusterId;
337  newCluster.addPixel(currentNeighborPixel);
338  }
339  }
340 
341  clusters_.push_back(std::move(newCluster));
342  ++clusterId;
343  }
344  }
