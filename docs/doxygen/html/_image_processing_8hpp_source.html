716  findNeighbors(inPixel, neighbors);

736  //=============================================================================
737  // Description:
740  void runClusterMaker()
741  {
742  uint32 clusterId = 0;
743 
744  for (uint32 xcdIdx = 0; xcdIdx < xcdsVec_.size(); ++xcdIdx)
745  {
746  Pixel<dtype>& currentPixel = xcdsVec_[xcdIdx];
747 
748  // not already visited
749  if (currentPixel.clusterId() == -1)
750  {
751  Cluster<dtype> newCluster(clusterId); // a new cluster
752  currentPixel.setClusterId(clusterId);
753  newCluster.addPixel(currentPixel); // assign pixel to cluster
754 
755  // get the neighbors
756  std::vector<uint32> neighborIds;
757  findNeighborXcds(currentPixel, neighborIds);
758  if (neighborIds.empty())
759  {
760  clusters_.push_back(newCluster);
761  ++clusterId;
762  continue;
763  }
764 
765  // loop through the neighbors
766  for (uint32 neighborsIdx = 0; neighborsIdx < neighborIds.size(); ++neighborsIdx)
767  {
768  Pixel<dtype>& currentNeighborPixel = xcdsVec_[neighborIds[neighborsIdx]];
769 
770  // go to neighbors
771  std::vector<uint32> newNeighborIds;
772  findNeighborXcds(currentNeighborPixel, newNeighborIds);
773 
774  // loop through the new neighbors and add them to neighbors
775  for (uint32 newNeighborsIdx = 0; newNeighborsIdx < newNeighborIds.size(); ++newNeighborsIdx)
776  {
777  // not already in neighbors
778  if (find(neighborIds.begin(), neighborIds.end(), newNeighborIds[newNeighborsIdx]) == neighborIds.end())
779  {
780  neighborIds.push_back(newNeighborIds[newNeighborsIdx]);
781  }
782  }
783 
784  // not already assigned to a cluster
785  if (currentNeighborPixel.clusterId() == -1)
786  {
787  currentNeighborPixel.setClusterId(clusterId);
788  newCluster.addPixel(currentNeighborPixel);
789  }
790  }
791 
792  clusters_.push_back(newCluster);
793  ++clusterId;
794  }
795  }

971  void centerOfMass(const Cluster<dtype>& inCluster)
972  {
973  Shape clusterShape(inCluster.height(), inCluster.width());
974  NdArray<dtype> clusterArray(clusterShape);
975  clusterArray.zeros();
976 
977  uint32 rowMin = inCluster.rowMin();
978  uint32 colMin = inCluster.colMin();
979  dtype intensity = inCluster.intensity();
980 
981  auto iter = inCluster.begin();
982  for (; iter < inCluster.end(); ++iter)
983  {
984  clusterArray(iter->row() - rowMin, iter->col() - colMin) = iter->intensity();
985  }
986 
987  // first get the row center
988  row_ = 0;
989  uint32 row = rowMin;
990  for (uint32 rowIdx = 0; rowIdx < clusterShape.rows; ++rowIdx)
991  {
992  double rowSum = 0;
993  for (uint32 colIdx = 0; colIdx < clusterShape.cols; ++colIdx)
994  {
995  rowSum += static_cast<double>(clusterArray(rowIdx, colIdx));
996  }
997  row_ += rowSum * static_cast<double>(row++);
998  }
999 
1000  row_ /= static_cast<double>(intensity);
1001 
1002  // then get the column center
1003  col_ = 0;
1004  uint32 col = colMin;
1005  for (uint32 colIdx = 0; colIdx < clusterShape.cols; ++colIdx)
1006  {
1007  double colSum = 0;
1008  for (uint32 rowIdx = 0; rowIdx < clusterShape.rows; ++rowIdx)
1009  {
1010  colSum += static_cast<double>(clusterArray(rowIdx, colIdx));
1011  }
1012  col_ += colSum * static_cast<double>(col++);
1013  }
1014 
1015  col_ /= static_cast<double>(intensity);
1016  }

1309  dtype generateThreshold(const NdArray<dtype>& inImageArray, double inRate)
1310  {
1311  if (inRate < 0 || inRate > 1)
1312  {
1313  std::string errStr = "ERROR: ImageProcessing::generateThreshold: input rate must be of the range [0, 1]";
1314  std::cerr << errStr << std::endl;
1315  throw std::invalid_argument(errStr);
1316  }
1317 
1318  // first build a histogram
1319  int32 minValue = static_cast<int32>(std::floor(inImageArray.min().item()));
1320  int32 maxValue = static_cast<int32>(std::floor(inImageArray.max().item()));
1321 
1322  if (inRate == 0)
1323  {
1324  return static_cast<dtype>(maxValue);
1325  }
1326  else if (inRate == 1)
1327  {
1328  if (DtypeInfo<dtype>::isSigned())
1329  {
1330  return static_cast<dtype>(minValue - 1);
1331  }
1332  else
1333  {
1334  return static_cast<dtype>(0);
1335  }
1336  }
1337 
1338  uint32 histSize = static_cast<uint32>(maxValue - minValue + 1);
1339 
1340  NdArray<double> histogram(1, histSize);
1341  histogram.zeros();
1342  uint32 numPixels = inImageArray.size();
1343  for (uint32 i = 0; i < numPixels; ++i)
1344  {
1345  uint32 bin = static_cast<uint32>(static_cast<int32>(std::floor(inImageArray[i])) - minValue);
1346  ++histogram[bin];
1347  }
1348 
1349  // integrate the normalized histogram from right to left to make a survival function (1 - CDF)
1350  double dNumPixels = static_cast<double>(numPixels);
1351  NdArray<double> survivalFunction(1, histSize + 1);
1352  survivalFunction[-1] = 0;
1353  for (int32 i = histSize - 1; i > -1; --i)
1354  {
1355  double histValue = histogram[i] / dNumPixels;
1356  survivalFunction[i] = survivalFunction[i + 1] + histValue;
1357  }
1358 
1359  // binary search through the survival function to find the rate
1360  uint32 indexLow = 0;
1361  uint32 indexHigh = histSize - 1;
1362  uint32 index = indexHigh / 2; // integer division
1363 
1364  bool keepGoing = true;
1365  while (keepGoing)
1366  {
1367  double value = survivalFunction[index];
1368  if (value < inRate)
1369  {
1370  indexHigh = index;
1371  }
1372  else if (value > inRate)
1373  {
1374  indexLow = index;
1375  }
1376  else
1377  {
1378  int32 thresh = static_cast<int32>(index) + minValue - 1;
1379  if (DtypeInfo<dtype>::isSigned())
1380  {
1381  return static_cast<dtype>(thresh);
1382  }
1383  else
1384  {
1385  return thresh < 0 ? 0 : static_cast<dtype>(thresh);
1386  }
1387  }
1388 
1389  if (indexHigh - indexLow < 2)
1390  {
1391  return static_cast<dtype>(static_cast<int32>(indexHigh) + minValue - 1);
1392  }
1393 
1394  index = indexLow + (indexHigh - indexLow) / 2;
1395  }
1396 
1397  // shouldn't ever get here but stop the compiler from throwing a warning
1398  return static_cast<dtype>(histSize - 1);
1399  }
