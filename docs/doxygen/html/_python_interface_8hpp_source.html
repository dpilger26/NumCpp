235  {
236  Shape inShape = inArray.shape();
237  std::vector<pybind11::ssize_t> shape{ inShape.rows, inShape.cols };
238  std::vector<pybind11::ssize_t> strides{ inShape.cols * sizeof(dtype), sizeof(dtype) };
239 
240  switch (returnPolicy)
241  {
242  case ReturnPolicy::COPY:
243  {
244  return pybind11::array_t<dtype>(shape, strides, inArray.data());
245  }
246  case ReturnPolicy::REFERENCE:
247  {
248  typename pybind11::capsule reference(inArray.data(), [](void* ptr) {});
249  return pybind11::array_t<dtype>(shape, strides, inArray.data(), reference);
250  }
251  case ReturnPolicy::TAKE_OWNERSHIP:
252  {
253  typename pybind11::capsule garbageCollect(inArray.dataRelease(),
254  [](void* ptr)
255  {
256  dtype* dataPtr = reinterpret_cast<dtype*>(ptr);
257  delete[] dataPtr;
258  }
259  );
260  return pybind11::array_t<dtype>(shape, strides, inArray.data(), garbageCollect);
261  }
262  default:
263  {
264  std::stringstream sstream;
265  sstream << "ReturnPolicy " << returnPolicyStringMap.at(returnPolicy) << " has not been implemented yet" << std::endl;
266  throw std::runtime_error(sstream.str());
267  }
