937  if (kernelSize % 2 == 0)
938  {

941

943

945  NdArray<double> kernel(kernelSize);
946  for (double row = 0; row < kernelSize; ++row)
947  {
948  for (double col = 0; col < kernelSize; ++col)

950  kernel(static_cast<uint32>(row), static_cast<uint32>(col)) = gaussian(row - kernalHalfSize, col - kernalHalfSize, inSigma);
951  }
952  }
953 
954  // normalize the kernel
955  kernel /= kernel.template sum<double>().item();
956 
957  // perform the convolution

961  inBoundaryType,
