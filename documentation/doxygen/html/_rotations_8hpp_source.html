129  // Method Description:
130  // returns a quaternion to rotate about the input axis by the input angle
131  //
133  // NdArray, x,y,z vector components
134  // angle in radians
136  // Quaternion
137  //
138  template<typename dtype>
139  static Quaternion angleAxisRotation(const NdArray<dtype>& inAxis, double inAngle)
140  {
141  if (inAxis.size() != 3)
142  {
143  throw std::invalid_argument("ERROR: Rotations::Quaternion::angleAxisRotation: input axis must be a cartesion vector of length = 3.");
144  }
145 
146  // normalize the input vector
147  NdArray<double> normAxis = inAxis.astype() / inAxis.norm().item();
148 
149  double i = static_cast<double>(normAxis[0]) * std::sin(inAngle / 2.0);
150  double j = static_cast<double>(normAxis[1]) * std::sin(inAngle / 2.0);
151  double k = static_cast<double>(normAxis[2]) * std::sin(inAngle / 2.0);
152  double s = std::cos(inAngle / 2.0);
153 
154  return Quaternion(i, j, k, s);
155  }

157  //============================================================================
158  // Method Description:
159  // angular velocity between the two quaternions. The norm
160  // of the array is the magnitude
161  //
163  // Quaternion 1
164  // Quaternion 2
165  // seperation time
167  // Quaternion
168  //
169  static NdArray<double> angularVelocity(const Quaternion& inQuat1, const Quaternion& inQuat2, double inTime)
170  {
171  NdArray<double> q0 = inQuat1.toNdArray();
172  NdArray<double> q1 = inQuat2.toNdArray();
173 
174  NdArray<double> qDot = q1 - q0;
175  qDot /= inTime;
176 
177  NdArray<double> eyeTimesScalar(3);
178  eyeTimesScalar.zeros();
179  eyeTimesScalar(0, 0) = inQuat2.s();
180  eyeTimesScalar(1, 1) = inQuat2.s();
181  eyeTimesScalar(2, 2) = inQuat2.s();
182 
183  NdArray<double> epsilonHat = Linalg<double>::hat(inQuat2.i(), inQuat2.j(), inQuat2.k());
184  NdArray<double> q(4, 3);
185  q.put(Slice(0, 3), Slice(0, 3), eyeTimesScalar + epsilonHat);
186  q(3, 0) = -inQuat2.i();
187  q(3, 1) = -inQuat2.j();
188  q(3, 2) = -inQuat2.k();
189 
190  NdArray<double> omega = q.transpose().dot<double>(qDot.transpose());
191  return std::move(omega *= 2);
192  }

295  //============================================================================
296  // Method Description:
297  // converts from a direction cosine matrix to a quaternion
298  //
300  // NdArray
302  // Quaternion
303  //
304  template<typename dtype>
305  static Quaternion fromDCM(const NdArray<dtype>& inDcm)
306  {
307  Shape inShape = inDcm.shape();
308  if (!(inShape.rows == 3 && inShape.cols == 3))
309  {
310  throw std::invalid_argument("ERROR: Rotations::Quaternion::fromDcm: input direction cosine matrix must have shape = (3,3).");
311  }
312 
313  NdArray<double> dcm = inDcm.astype();
314 
315  NdArray<double> checks(1, 4);
316  checks[0] = dcm(0, 0) + dcm(1, 1) + dcm(2, 2);
317  checks[1] = dcm(0, 0) - dcm(1, 1) + dcm(2, 2);
318  checks[2] = dcm(0, 0) - dcm(1, 1) - dcm(2, 2);
319  checks[3] = dcm(0, 0) + dcm(1, 1) - dcm(2, 2);
320 
321  uint32 maxIdx = Methods<double>::argmax(checks).item();
322 
323  double q0 = 0;
324  double q1 = 0;
325  double q2 = 0;
326  double q3 = 0;
327 
328  switch (maxIdx)
329  {
330  case 0:
331  {
332  q3 = 0.5 * std::sqrt(1 + dcm(0, 0) + dcm(1, 1) + dcm(2, 2));
333  q0 = (dcm(1, 2) - dcm(2, 1)) / (4 * q3);
334  q1 = (dcm(2, 0) - dcm(0, 2)) / (4 * q3);
335  q2 = (dcm(0, 1) - dcm(1, 0)) / (4 * q3);
336 
337  break;
338  }
339  case 1:
340  {
341  q2 = 0.5 * std::sqrt(1 - dcm(0, 0) - dcm(1, 1) + dcm(2, 2));
342  q0 = (dcm(0, 2) + dcm(2, 0)) / (4 * q2);
343  q1 = (dcm(1, 2) + dcm(2, 1)) / (4 * q2);
344  q3 = (dcm(0, 1) - dcm(1, 0)) / (4 * q2);
345 
346  break;
347  }
348  case 2:
349  {
350  q0 = 0.5 * std::sqrt(1 + dcm(0, 0) - dcm(1, 1) - dcm(2, 2));
351  q1 = (dcm(0, 1) + dcm(1, 0)) / (4 * q0);
352  q2 = (dcm(0, 2) + dcm(2, 0)) / (4 * q0);
353  q3 = (dcm(1, 2) - dcm(2, 1)) / (4 * q0);
354 
355  break;
356  }
357  case 3:
358  {
359  q1 = 0.5 * std::sqrt(1 - dcm(0, 0) + dcm(1, 1) - dcm(2, 2));
360  q0 = (dcm(0, 1) + dcm(1, 0)) / (4 * q1);
361  q2 = (dcm(1, 2) + dcm(2, 1)) / (4 * q1);
362  q3 = (dcm(2, 0) - dcm(0, 2)) / (4 * q1);
363 
364  break;
365  }
366  }
367 
368  return Quaternion(q0, q1, q2, q3);
369  }

372  // Method Description:
373  // linearly interpolates between the two quaternions
374  //
376  // Quaternion 1
377  // Quaternion 2
378  // percent [0, 1]
380  // Quaternion
381  //
382  static Quaternion nlerp(const Quaternion& inQuat1, const Quaternion& inQuat2, double inPercent)
383  {
384  if (inPercent < 0 || inPercent > 1)
385  {
386  throw std::invalid_argument("ERROR: Rotations::Quaternion::nlerp: input percent must be of the range [0,1].");
387  }
388 
389  if (inPercent == 0)
390  {
391  return inQuat1;
392  }
393  else if (inPercent == 1)
394  {
395  return inQuat2;
396  }
397 
398  double oneMinus = 1.0 - inPercent;
399  double i = oneMinus * inQuat1.data_[0] + inPercent * inQuat2.data_[0];
400  double j = oneMinus * inQuat1.data_[1] + inPercent * inQuat2.data_[1];
401  double k = oneMinus * inQuat1.data_[2] + inPercent * inQuat2.data_[2];
402  double s = oneMinus * inQuat1.data_[3] + inPercent * inQuat2.data_[3];
403 
404  return Quaternion(i, j, k, s);
405  }

470  //============================================================================
471  // Method Description:
472  // spherical linear interpolates between the two quaternions
473  //
475  // Quaternion 1
476  // Quaternion 2
477  // percent (0, 1)
479  // Quaternion
480  //
481  static Quaternion slerp(const Quaternion& inQuat1, const Quaternion& inQuat2, double inPercent)
482  {
483  if (inPercent < 0 || inPercent > 1)
484  {
485  throw std::invalid_argument("ERROR: Rotations::Quaternion::slerp: input percent must be of the range [0, 1]");
486  }
487 
488  if (inPercent == 0)
489  {
490  return inQuat1;
491  }
492  else if (inPercent == 1)
493  {
494  return inQuat2;
495  }
496 
497  double dotProduct = Methods<double>::dot<double>(inQuat1.toNdArray(), inQuat2.toNdArray()).item();
498 
499  // If the dot product is negative, the quaternions
500  // have opposite handed-ness and slerp won't take
501  // the shorter path. Fix by reversing one quaternion.
502  Quaternion quat1Copy(inQuat1);
503  if (dotProduct < 0.0)
504  {
505  quat1Copy *= -1;
506  dotProduct *= -1;
507  }
508 
509  const double DOT_THRESHOLD = 0.9995;
510  if (dotProduct > DOT_THRESHOLD) {
511  // If the inputs are too close for comfort, linearly interpolate
512  // and normalize the result.
513  return nlerp(inQuat1, inQuat2, inPercent);
514  }
515 
516  dotProduct = Methods<double>::clip(dotProduct, -1.0, 1.0); // Robustness: Stay within domain of acos()
517  double theta0 = std::acos(dotProduct); // angle between input vectors
518  double theta = theta0 * inPercent; // angle between v0 and result
519 
520  double s0 = std::cos(theta) - dotProduct * std::sin(theta) / std::sin(theta0); // == sin(theta_0 - theta) / sin(theta_0)
521  double s1 = std::sin(theta) / std::sin(theta0);
522 
523  NdArray<double> interpQuat = (inQuat1.toNdArray() * s0) + (inQuat2.toNdArray() * s1);
524  return Quaternion(interpQuat);
525  }

559  //============================================================================
560  // Method Description:
561  // returns the direction cosine matrix
562  //
564  // None
566  // NdArray
567  //
568  NdArray<double> toDCM() const
569  {
570  NdArray<double> dcm(3);
571 
572  double q0 = i();
573  double q1 = j();
574  double q2 = k();
575  double q3 = s();
576 
577  dcm(0, 0) = Utils<double>::sqr(q3) + Utils<double>::sqr(q0) - Utils<double>::sqr(q1) - Utils<double>::sqr(q2);
578  dcm(0, 1) = 2 * (q0 * q1 + q3 * q2);
579  dcm(0, 2) = 2 * (q0 * q2 - q3 * q1);
580  dcm(1, 0) = 2 * (q0 * q1 - q3 * q2);
581  dcm(1, 1) = Utils<double>::sqr(q3) - Utils<double>::sqr(q0) + Utils<double>::sqr(q1) - Utils<double>::sqr(q2);;
582  dcm(1, 2) = 2 * (q1 * q2 + q3 * q0);
583  dcm(2, 0) = 2 * (q0 * q2 + q3 * q1);
584  dcm(2, 1) = 2 * (q1 * q2 - q3 * q0);
585  dcm(2, 2) = Utils<double>::sqr(q3) - Utils<double>::sqr(q0) - Utils<double>::sqr(q1) + Utils<double>::sqr(q2);;
586 
587  return std::move(dcm);
588  }

795  //============================================================================
796  // Method Description:
797  // multiplication assignment operator
798  //
800  // Quaternion
802  // Quaternion
803  //
804  Quaternion& operator*=(const Quaternion& inRhs)
805  {
