187  //============================================================================
188  // Method Description:
189  // Constructor
190  //
192  // 2D initializer list
194  // None
195  //
196  NdArray(const std::initializer_list<std::initializer_list<dtype> >& inList) :
197  shape_(static_cast<uint32>(inList.size()), 0),
198  size_(0),
199  endianess_(Endian::NATIVE),
200  array_(nullptr)
201  {
202  typename std::initializer_list<std::initializer_list<dtype> >::iterator iter;
203  for (iter = inList.begin(); iter < inList.end(); ++iter)
204  {
205  size_ += static_cast<uint32>(iter->size());
206 
207  if (shape_.cols == 0)
208  {
209  shape_.cols = static_cast<uint32>(iter->size());
210  }
211  else if (iter->size() != shape_.cols)
212  {
213  throw std::runtime_error("ERROR: NdArray::Constructor: All rows of the initializer list needs to have the same number of elements");
214  }
215  }
216 
217  array_ = new dtype[size_];
218  uint16 row = 0;
219  for (iter = inList.begin(); iter < inList.end(); ++iter)
220  {
221  std::copy(iter->begin(), iter->end(), array_ + row * shape_.cols);
222  ++row;
223  }
224  }

489  // 1D Slicing access operator with no bounds checking.
490  // returned array is of the range [start, stop).
491  //
493  // Slice
495  // NdArray
496  //

498  {
499  Slice inSliceCopy(inSlice);
500 
501  uint32 counter = 0;
502  NdArray<dtype> returnArray(1, inSliceCopy.numElements(size_));
503  for (int32 i = inSliceCopy.start; i < inSliceCopy.stop; i += inSliceCopy.step)
504  {
505  returnArray[counter++] = this->at(i);
506  }
507 
508  return std::move(returnArray);
509  }

539  }
