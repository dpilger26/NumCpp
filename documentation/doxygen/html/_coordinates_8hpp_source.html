567  //============================================================================
568  // Method Description:
569  // converts polar coordinates to cartesian coordinates
570  //
572  // None
574  // None
575  //
576  void cartesianToPolar()
577  {
578  dtype degreesRa = static_cast<dtype>(Methods<dtype>::rad2deg(std::atan2(y_, x_)));
579  if (degreesRa < 0)
580  {
581  degreesRa += 360;
582  }
583  ra_ = RA<dtype>(degreesRa);
584 
585  double r = std::sqrt(static_cast<double>(Utils<dtype>::sqr(x_)) + static_cast<double>(Utils<dtype>::sqr(y_)) + static_cast<double>(Utils<dtype>::sqr(z_)));
586  dtype degreesDec = static_cast<dtype>(Methods<double>::rad2deg(std::asin(static_cast<double>(z_) / r)));
587  dec_ = Dec<dtype>(degreesDec);
588  }
589 
590  //============================================================================
591  // Method Description:
592  // converts polar coordinates to cartesian coordinates
593  //
595  // None
597  // None
598  //
599  void polarToCartesian()
600  {
601  double raRadians = Methods<double>::deg2rad(static_cast<double>(ra_.degrees()));
602  double decRadians = Methods<double>::deg2rad(static_cast<double>(dec_.degrees()));
603 
604  x_ = static_cast<dtype>(std::cos(raRadians) * std::cos(decRadians));
605  y_ = static_cast<dtype>(std::sin(raRadians) * std::cos(decRadians));
606  z_ = static_cast<dtype>(std::sin(decRadians));
607  }
