746  // Description:
749  void runClusterMaker()
750  {
751  uint32 clusterId = 0;
752 
753  for (uint32 xcdIdx = 0; xcdIdx < xcdsVec_.size(); ++xcdIdx)
754  {
755  Pixel& currentPixel = xcdsVec_[xcdIdx];
756 
757  // not already visited
758  if (currentPixel.clusterId() == -1)
759  {
760  Cluster newCluster(clusterId); // a new cluster
761  currentPixel.setClusterId(clusterId);
762  newCluster.addPixel(currentPixel); // assign pixel to cluster
763 
764  // get the neighbors
765  std::vector<uint32> neighborIds;
766  findNeighborXcds(currentPixel, neighborIds);
767  if (neighborIds.empty())
768  {
769  clusters_.push_back(newCluster);
770  ++clusterId;
771  continue;
772  }
773 
774  // loop through the neighbors
775  for (uint32 neighborsIdx = 0; neighborsIdx < neighborIds.size(); ++neighborsIdx)
776  {
777  Pixel& currentNeighborPixel = xcdsVec_[neighborIds[neighborsIdx]];
778 
779  // go to neighbors
780  std::vector<uint32> newNeighborIds;
781  findNeighborXcds(currentNeighborPixel, newNeighborIds);
782 
783  // loop through the new neighbors and add them to neighbors
784  for (uint32 newNeighborsIdx = 0; newNeighborsIdx < newNeighborIds.size(); ++newNeighborsIdx)
785  {
786  // not already in neighbors
787  if (find(neighborIds.begin(), neighborIds.end(), newNeighborIds[newNeighborsIdx]) == neighborIds.end())
788  {
789  neighborIds.push_back(newNeighborIds[newNeighborsIdx]);
790  }
791  }
792 
793  // not already assigned to a cluster
794  if (currentNeighborPixel.clusterId() == -1)
795  {
796  currentNeighborPixel.setClusterId(clusterId);
797  newCluster.addPixel(currentNeighborPixel);
798  }
799  }
800 
801  clusters_.push_back(newCluster);
802  ++clusterId;
803  }
804  }

1015  colSum += static_cast<double>(clusterArray(rowIdx, colIdx));

1296  static dtype generateThreshold(const NdArray<dtype>& inImageArray, double inRate)
1297  {
1298  if (inRate < 0 || inRate > 1)
1299  {
1300  throw std::invalid_argument("ERROR: ImageProcessing::generateThreshold: input rate must be of the range [0, 1]");
1301  }
1302 
1303  // first build a histogram
1304  int32 minValue = static_cast<int32>(std::floor(inImageArray.min().item()));
1305  int32 maxValue = static_cast<int32>(std::floor(inImageArray.max().item()));
1306 
1307  if (inRate == 0)
1308  {
1309  return static_cast<dtype>(maxValue);
1310  }
1311  else if (inRate == 1)
1312  {
1313  if (DtypeInfo<dtype>::isSigned())
1314  {
1315  return static_cast<dtype>(minValue - 1);
1316  }
1317  else
1318  {
1319  return static_cast<dtype>(0);
1320  }
1321  }
1322 
1323  uint32 histSize = static_cast<uint32>(maxValue - minValue + 1);
1324 
1325  NdArray<double> histogram(1, histSize);
1326  histogram.zeros();
1327  uint32 numPixels = inImageArray.size();
1328  for (uint32 i = 0; i < numPixels; ++i)
1329  {
1330  uint32 bin = static_cast<uint32>(static_cast<int32>(std::floor(inImageArray[i])) - minValue);
1331  ++histogram[bin];
1332  }
1333 
1334  // integrate the normalized histogram from right to left to make a survival function (1 - CDF)
1335  double dNumPixels = static_cast<double>(numPixels);
1336  NdArray<double> survivalFunction(1, histSize + 1);
1337  survivalFunction[-1] = 0;
1338  for (int32 i = histSize - 1; i > -1; --i)
1339  {
1340  double histValue = histogram[i] / dNumPixels;
1341  survivalFunction[i] = survivalFunction[i + 1] + histValue;
1342  }
1343 
1344  // binary search through the survival function to find the rate
1345  uint32 indexLow = 0;
1346  uint32 indexHigh = histSize - 1;
1347  uint32 index = indexHigh / 2; // integer division
1348 
1349  bool keepGoing = true;
1350  while (keepGoing)
1351  {
1352  double value = survivalFunction[index];
1353  if (value < inRate)
1354  {
1355  indexHigh = index;
1356  }
1357  else if (value > inRate)
1358  {
1359  indexLow = index;
1360  }
1361  else
1362  {
1363  int32 thresh = static_cast<int32>(index) + minValue - 1;
1364  if (DtypeInfo<dtype>::isSigned())
1365  {
1366  return static_cast<dtype>(thresh);
1367  }
1368  else
1369  {
1370  return thresh < 0 ? 0 : static_cast<dtype>(thresh);
1371  }
1372  }
1373 
1374  if (indexHigh - indexLow < 2)
1375  {
1376  return static_cast<dtype>(static_cast<int32>(indexHigh) + minValue - 1);
1377  }
1378 
1379  index = indexLow + (indexHigh - indexLow) / 2;
1380  }
1381 
1382  // shouldn't ever get here but stop the compiler from throwing a warning
1383  return static_cast<dtype>(histSize - 1);
1384  }
