530  {
531  dtype degreesRa = static_cast<dtype>(Methods<dtype>::rad2deg(std::atan2(y_, x_)));
532  if (degreesRa < 0)
533  {
534  degreesRa += 360;
535  }
536  ra_ = RA<dtype>(degreesRa);
537 
538  double r = std::sqrt(static_cast<double>(Utils<dtype>::sqr(x_)) + static_cast<double>(Utils<dtype>::sqr(y_)) + static_cast<double>(Utils<dtype>::sqr(z_)));
539  dtype degreesDec = static_cast<dtype>(Methods<double>::rad2deg(std::asin(static_cast<double>(z_) / r)));
540  dec_ = Dec<dtype>(degreesDec);
541  }
