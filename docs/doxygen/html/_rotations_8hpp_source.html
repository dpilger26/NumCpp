161  static NdArray<double> angularVelocity(const Quaternion& inQuat1, const Quaternion& inQuat2, double inTime)
162  {
163  NdArray<double> q0 = inQuat1.toNdArray();
164  NdArray<double> q1 = inQuat2.toNdArray();
165 
166  NdArray<double> qDot = q1 - q0;
167  qDot /= inTime;
168 
169  NdArray<double> eyeTimesScalar(3);
170  eyeTimesScalar.zeros();
171  eyeTimesScalar(0, 0) = inQuat2.s();
172  eyeTimesScalar(1, 1) = inQuat2.s();
173  eyeTimesScalar(2, 2) = inQuat2.s();
174 
175  NdArray<double> epsilonHat = Linalg<double>::hat(inQuat2.i(), inQuat2.j(), inQuat2.k());
176  NdArray<double> q(4, 3);
177  q.put(Slice(0, 3), Slice(0, 3), eyeTimesScalar + epsilonHat);
178  q(3, 0) = -inQuat2.i();
179  q(3, 1) = -inQuat2.j();
180  q(3, 2) = -inQuat2.k();
181 
182  NdArray<double> omega = q.transpose().dot<double>(qDot.transpose());
183  return std::move(omega *= 2);
184  }

284  static Quaternion fromDCM(const NdArray<dtype>& inDcm)
285  {
286  Shape inShape = inDcm.shape();
287  if (!(inShape.rows == 3 && inShape.cols == 3))
288  {
289  throw std::invalid_argument("ERROR: Rotations::Quaternion::fromDcm: input direction cosine matrix must have shape = (3,3).");
290  }
291 
292  NdArray<double> dcm = inDcm.astype();
293 
294  NdArray<double> checks(1, 4);
295  checks[0] = dcm(0, 0) + dcm(1, 1) + dcm(2, 2);
296  checks[1] = dcm(0, 0) - dcm(1, 1) + dcm(2, 2);
297  checks[2] = dcm(0, 0) - dcm(1, 1) - dcm(2, 2);
298  checks[3] = dcm(0, 0) + dcm(1, 1) - dcm(2, 2);
299 
300  uint32 maxIdx = Methods<double>::argmax(checks).item();
301 
302  double q0 = 0;
303  double q1 = 0;
304  double q2 = 0;
305  double q3 = 0;
306 
307  switch (maxIdx)
308  {
309  case 0:
310  {
311  q3 = 0.5 * std::sqrt(1 + dcm(0, 0) + dcm(1, 1) + dcm(2, 2));
312  q0 = (dcm(1, 2) - dcm(2, 1)) / (4 * q3);
313  q1 = (dcm(2, 0) - dcm(0, 2)) / (4 * q3);
314  q2 = (dcm(0, 1) - dcm(1, 0)) / (4 * q3);
315 
316  break;
317  }
318  case 1:
319  {
320  q2 = 0.5 * std::sqrt(1 - dcm(0, 0) - dcm(1, 1) + dcm(2, 2));
321  q0 = (dcm(0, 2) + dcm(2, 0)) / (4 * q2);
322  q1 = (dcm(1, 2) + dcm(2, 1)) / (4 * q2);
323  q3 = (dcm(0, 1) - dcm(1, 0)) / (4 * q2);
324 
325  break;
326  }
327  case 2:
328  {
329  q0 = 0.5 * std::sqrt(1 + dcm(0, 0) - dcm(1, 1) - dcm(2, 2));
330  q1 = (dcm(0, 1) + dcm(1, 0)) / (4 * q0);
331  q2 = (dcm(0, 2) + dcm(2, 0)) / (4 * q0);
332  q3 = (dcm(1, 2) - dcm(2, 1)) / (4 * q0);
333 
334  break;
335  }
336  case 3:
337  {
338  q1 = 0.5 * std::sqrt(1 - dcm(0, 0) + dcm(1, 1) - dcm(2, 2));
339  q0 = (dcm(0, 1) + dcm(1, 0)) / (4 * q1);
340  q2 = (dcm(1, 2) + dcm(2, 1)) / (4 * q1);
341  q3 = (dcm(2, 0) - dcm(0, 2)) / (4 * q1);
342 
343  break;
344  }
345  }
346 
347  return Quaternion(q0, q1, q2, q3);
348  }

450  static Quaternion slerp(const Quaternion& inQuat1, const Quaternion& inQuat2, double inPercent)
451  {
452  if (inPercent < 0 || inPercent > 1)
453  {
454  throw std::invalid_argument("ERROR: Rotations::Quaternion::slerp: input percent must be of the range [0, 1]");
455  }
456 
457  if (inPercent == 0)
458  {
459  return inQuat1;
460  }
461  else if (inPercent == 1)
462  {
463  return inQuat2;
464  }
465 
466  double dotProduct = Methods<double>::dot<double>(inQuat1.toNdArray(), inQuat2.toNdArray()).item();
467 
468  // If the dot product is negative, the quaternions
469  // have opposite handed-ness and slerp won't take
470  // the shorter path. Fix by reversing one quaternion.
471  Quaternion quat1Copy(inQuat1);
472  if (dotProduct < 0.0)
473  {
474  quat1Copy *= -1;
475  dotProduct *= -1;
476  }
477 
478  const double DOT_THRESHOLD = 0.9995;
479  if (dotProduct > DOT_THRESHOLD) {
480  // If the inputs are too close for comfort, linearly interpolate
481  // and normalize the result.
482  return nlerp(inQuat1, inQuat2, inPercent);
483  }
484 
485  dotProduct = Methods<double>::clip(dotProduct, -1.0, 1.0); // Robustness: Stay within domain of acos()
486  double theta0 = std::acos(dotProduct); // angle between input vectors
487  double theta = theta0 * inPercent; // angle between v0 and result
488 
489  double s0 = std::cos(theta) - dotProduct * std::sin(theta) / std::sin(theta0); // == sin(theta_0 - theta) / sin(theta_0)
490  double s1 = std::sin(theta) / std::sin(theta0);
491 
492  NdArray<double> interpQuat = (inQuat1.toNdArray() * s0) + (inQuat2.toNdArray() * s1);
493  return Quaternion(interpQuat);
494  }

538  double q2 = k();

541  dcm(0, 0) = Utils<double>::sqr(q3) + Utils<double>::sqr(q0) - Utils<double>::sqr(q1) - Utils<double>::sqr(q2);

547  dcm(2, 0) = 2 * (q0 * q2 + q3 * q1);
548  dcm(2, 1) = 2 * (q1 * q2 - q3 * q0);
549  dcm(2, 2) = Utils<double>::sqr(q3) - Utils<double>::sqr(q0) - Utils<double>::sqr(q1) + Utils<double>::sqr(q2);;
